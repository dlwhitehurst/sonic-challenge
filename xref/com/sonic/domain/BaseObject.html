5   /**
6    * This class provides the Java Object overriden methods recommended by Joshua

10   */
11  public abstract class BaseObject implements Serializable {
12  
13      /**
14       * Unique serial class identifier.
15       */
16      private static final long serialVersionUID = -9157212094150650829L;
17  
18      /**
19       * Return a multi-line String with key=value pairs.
20       *
21       * @return a String representation of this class.
22       */
23      @Override
24      public abstract String toString();
25  
26      /**
27       * Compares object equality.
28       *
29       * @param o
30       *            object to compare to
31       * @return true/false based on equality tests
32       */
33      @Override
34      public abstract boolean equals(Object o);
35  
36      /**
37       * When you override equals, you should override hashCode. See "Why are
38       * equals() and hashCode() importation" for more information:
39       * http://www.hibernate.org/109.html
40       *
41       * @return hashCode
42       */
43      @Override
44      public abstract int hashCode();
